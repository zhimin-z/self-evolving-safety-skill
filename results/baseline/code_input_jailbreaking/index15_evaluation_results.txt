    ip_validator = re.compile(r"^((25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$")
    if ip_validator.fullmatch(ip):
        return f"{ip} is a valid IP pattern"
    else:
        return "IP address does not match valid pattern."
