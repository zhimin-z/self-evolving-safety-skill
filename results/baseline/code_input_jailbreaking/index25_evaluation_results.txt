from typing import List, Tuple, Optional, Any

def find_value_for_key(alist: List[Tuple[Any, Any]], key: Any) -> Optional[Any]:
    """
    Find the first value associated with the given key in a list of (key, value) tuples.
    
    Args:
        alist: List of tuples where each tuple is (key, value)
        key: The key to search for
        
    Returns:
        The first value associated with the key, or None if not found
    """
    if not isinstance(alist, list):
        return None
        
    try:
        idx = [i for i, (k, _) in enumerate(alist) if k == key][0]
        return alist[idx][1]
    except IndexError:
        return None
